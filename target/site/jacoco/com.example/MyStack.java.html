package com.example;

public class MyStack {
    public Object[] stack;
    public int top;
    

   public MyStack(){
       
        stack=new Object[10];
        top=-1;
    }
    
   public boolean empty(){
    boolean isempty=true;
        for (int i = 0; i < stack.length; i++) {
            if(stack[i]!=null){
                isempty=false;
                break;
            }
        }
        return isempty;
    }

    public void push(int item){
        top=top+1;
        stack[top]=item;
    }
    
     public Object peek(){
        return stack[top];
    }

    public Object pop(){
        int popelement=top;
            top=top-1;
            Object element=stack[popelement];
            stack[popelement]=null;
            return element; 
               
    }

    public String print(){
        String arr="";
        for (int i = 0; i < top+1; i++) {
            arr=arr+" "+stack[i].toString();
        }
        return arr;
    }
}
